Servicios que utiliza esta aplicación. Se configuran mediante variables de entorno en el servidor.

🟩
Supabase — Base de datos
Base de datos PostgreSQL en la nube. Almacena usuarios, placas y alertas. Configurado con SUPABASE_URL y SUPABASE_KEY.
ACTIVO
🤖
CallMeBot — Notificaciones WhatsApp
Envía alertas por WhatsApp de forma gratuita cuando se detecta una notificación ANM. Configurado con CALLMEBOT_KEY.
ACTIVO
🐙
GitHub Actions — Consultas automáticas
Ejecuta la consulta diaria a la ANM todos los días a las 6:00 a.m. (hora Colombia). No requiere servidor adicional. Configurado en .github/workflows/consulta-diaria.yml.
ACTIVO
SQL — Crear todas las tablas
-- Usuarios
CREATE TABLE usuarios (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  nombre text NOT NULL,
  telefono text NOT NULL UNIQUE,
  password_hash text NOT NULL,
  creado_en timestamptz DEFAULT now()
);

-- Códigos de verificación WhatsApp
CREATE TABLE codigos_verificacion (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  telefono text NOT NULL,
  codigo text NOT NULL,
  expira_en timestamptz NOT NULL,
  usado boolean DEFAULT false
);

-- Placas (por usuario)
CREATE TABLE placas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  usuario_id uuid REFERENCES usuarios(id) ON DELETE CASCADE,
  placa text NOT NULL,
  nombre text NOT NULL,
  celular text NOT NULL,
  estado text DEFAULT 'activa',
  ultima_consulta timestamptz,
  fecha_notificacion text,
  creado_en timestamptz DEFAULT now(),
  UNIQUE(usuario_id, placa)
);

-- Alertas (por usuario)
CREATE TABLE alertas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  usuario_id uuid REFERENCES usuarios(id) ON DELETE CASCADE,
  placa text NOT NULL,
  nombre text NOT NULL,
  celular text NOT NULL,
  fecha_publicacion text,
  mensaje text,
  creado_en timestamptz DEFAULT now()
);
